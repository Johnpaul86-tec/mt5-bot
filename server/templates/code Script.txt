    global running
    while running:
        signal = analyze_market("EURUSD")

        if signal == "BUY":
            place_trade("EURUSD", "buy")
            log("BUY trade executed")

        elif signal == "SELL":

    global running
    if not running:
        running = True
        thread = threading.Thread(target=trading_loop)

import MetaTrader5 as mt5
import pandas as pd

def get_data(symbol, timeframe=mt5.TIMEFRAME_M5, bars=100):
    rates = mt5.copy_rates_from_pos(symbol, timeframe, 0, bars)
    return pd.DataFrame(rates)

def detect_trend(df):
    if df['close'].iloc[-1] > df['close'].mean():
        return "UP"
    else:
        return "DOWN"

def break_of_structure(df):
    return df['high'].iloc[-1] > df['high'].iloc[-2]

def change_of_character(df):
    return df['low'].iloc[-1] < df['low'].iloc[-2]

def order_block(df):
    return abs(df['open'].iloc[-2] - df['close'].iloc[-2]) < 0.0005

def confirmation(df):
    return df['close'].iloc[-1] > df['open'].iloc[-1]

def analyze_market(symbol):
    if not mt5.initialize():
        return None

    df = get_data(symbol)

    trend = detect_trend(df)
    bos = break_of_structure(df)
    choch = change_of_character(df)
    ob = order_block(df)
    confirm = confirmation(df)

    mt5.shutdown()

    if trend == "UP" and bos and ob and confirm:
        return "BUY"

    if trend == "DOWN" and choch and ob and not confirm:
        return "SELL"

    return None

def get_account_info():
    if not mt5.initialize():
        return {}

    info = mt5.account_info()
    mt5.shutdown()

    if info:
        return info._asdict()
    return {}

def place_trade(symbol, order_type):
    if not mt5.initialize():
        return False

    lot = 0.1
    price = mt5.symbol_info_tick(symbol).ask

    order = {
        "action": mt5.TRADE_ACTION_DEAL,
        "symbol": symbol,
        "volume": lot,
        "type": mt5.ORDER_TYPE_BUY if order_type == "buy" else mt5.ORDER_TYPE_SELL,
        "price": price,
        "deviation": 10,
        "magic": 123456,
        "comment": "Bot Trade",
        "type_time": mt5.ORDER_TIME_GTC,
        "type_filling": mt5.ORDER_FILLING_IOC,
    }

    result = mt5.order_send(order)
    mt5.shutdown()
    return result

def login_required(func):
    from functools import wraps
    def wrapper(*args, **kwargs):
        if "user_id" not in session:
            return redirect("/login")
        return func(*args, **kwargs)
    return wraps(func)(wrapper)

def place_trade(symbol, action, lot, sl, tp):
    import MetaTrader5 as mt5

    price = mt5.symbol_info_tick(symbol).ask

    request = {
        "action": mt5.TRADE_ACTION_DEAL,
        "symbol": symbol,
        "volume": lot,
        "type": mt5.ORDER_TYPE_BUY if action == "BUY" else mt5.ORDER_TYPE_SELL,
        "price": price,
        "sl": price - sl * 0.0001,
        "tp": price + tp * 0.0001,
        "deviation": 10,
        "magic": 123456,
        "comment": "AI Bot",
        "type_time": mt5.ORDER_TIME_GTC,
        "type_filling": mt5.ORDER_FILLING_IOC,
    }

    return mt5.order_send(request)
